SELECT approx_percentile(spread_bp, 0.5, 4) AS median_bp, approx_percentile(spread_bp, 0.05, 4) AS p5_bp, approx_percentile(spread_bp, 0.95, 4) AS p95_bp, count() AS n FROM (  SELECT (ask_price - bid_price) / ((ask_price + bid_price) / 2.0) * 1e4 AS spread_bp   FROM pancake_bookticker   WHERE symbol = 'SOLUSDT'   AND timestamp > dateadd('h', -1, now())   AND bid_price > 0 AND ask_price > 0)
